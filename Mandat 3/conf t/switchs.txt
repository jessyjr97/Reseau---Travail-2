!Switch-Racine

enable
  configure terminal
	vlan 110
		name Directeurs
	vlan 120
		name Vendeurs
	vlan 130
		name Service
	vlan 140
		name Peinture
	vlan 150
		name Reception
	vlan 160
		name AP-Employes
	vlan 170
		name AP-Clients
    vlan 180
		name Tech
    exit
	interface f0/1
		no shut
		switchport mode access
		switchport access vlan 180
	interface f0/2
		no shut
		switchport mode access
		switchport access vlan 170
	interface f0/3
		no shut
		switchport mode access
		switchport access vlan 160
	interface range f0/4-17
		shut
	interface f0/18
		no shut
        no switchport trunk allowed vlan
        switchport mode trunk
        switchport trunk allowed vlan 110,130,140
	interface f0/19
		no shut
		switchport mode trunk
        switchport trunk allowed vlan 120,150
	interface f0/20
		no shut
		switchport mode trunk
		switchport access vlan 110
	interface range f0/21 - 23
		no shut
    interface f0/24
		no shut
		no switchport trunk allowed vlan
		switchport mode trunk
		switchport trunk allowed vlan 1,110,120,130,140,150,160,170,180
    exit
  exit
  copy running-config startup-config

exit
